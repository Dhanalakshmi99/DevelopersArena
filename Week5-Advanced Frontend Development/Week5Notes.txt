1. CSS Grid and Flexbox Mastery
Definition

CSS Grid is a 2-dimensional layout system (rows + columns).
Flexbox is a 1-dimensional layout system (row or column).

Examples
CSS Grid Example
.grid {
  display: grid;
  grid-template-columns: repeat(3, 1fr);
  gap: 20px;
}

Flexbox Example
.flex {
  display: flex;
  justify-content: space-between;
  align-items: center;
}

2. CSS Animations and Transitions
Definition

Transitions animate changes between one state and another (e.g. hover).

Animations use keyframes to create continuous or more complex movement.

Examples
Transition Example
button {
  background: #007bff;
  transition: background 0.3s ease;
}

button:hover {
  background: #0056b3;
}

Animation Example
.box {
  width: 50px;
  height: 50px;
  background: red;
  animation: move 2s infinite alternate;
}

@keyframes move {
  from { transform: translateX(0); }
  to   { transform: translateX(100px); }
}

3. JavaScript ES6+ Features
Definition

ES6+ refers to modern JavaScript features introduced after 2015 that make code cleaner, faster, and more powerful.

Examples
// Let & Const
const pi = 3.14;

// Arrow functions
const greet = name => `Hello, ${name}`;

// Template literals
console.log(`2 + 2 = ${2 + 2}`);

// Destructuring
const user = { name: "Alex", age: 21 };
const { name, age } = user;

// Spread operator
const arr = [1, 2, 3];
const newArr = [...arr, 4, 5];

    },
    {

        name: "Blue Jeans",
        price: 1099.99,
        image: "https://production-inventory-service.s3.ap-south-1.amazonaws.com/kodnest-documents/salessavvy/COSTOMER+IMAGES/pant1.png"
    },

        id: 4,

        price: 24999.99,

export default function App() {
  const [cart, setCart] = useState([]);
  const [open, setOpen] = useState(false);

  // Add item to cart
  const addToCart = (id) => {
    const product = products.find((p) => p.id === id);
    const exists = cart.find((c) => c.id === id);

    if (exists) {
      setCart(
        cart.map((item) =>
          item.id === id ? { ...item, qty: item.qty + 1 } : item
        )
      );
    } else {
      setCart([...cart, { ...product, qty: 1 }]);
    }
  };

  // Remove item
  const removeItem = (id) => {
    setCart(cart.filter((item) => item.id !== id));
  };

  // Total price
  const total = cart.reduce((sum, i) => sum + i.price * i.qty, 0);

  return (
    <div>
      {/* Header */}
      <header className="header">
        <h1>ShopNow</h1>
        <button className="cart-btn" onClick={() => setOpen(!open)}>
          Cart ({cart.length})
        </button>
      </header>

      {/* Product List */}
      <div className="products">
        {products.map((p) => (
          <div className="product" key={p.id}>
            <img src={p.image} alt={p.name} />
            <h3>{p.name}</h3>
            <p>{p.price.toFixed(2)}</p>
            <button onClick={() => addToCart(p.id)}>Add to Cart</button>
          </div>
        ))}
      </div>

      {/* Cart Drawer */}
      <div className={`cart ${open ? "open" : ""}`}>
        <h2>Your Cart</h2>

        <div className="cart-items">
          {cart.map((item) => (
            <div className="cart-item" key={item.id}>
              <span>
                {item.name} x {item.qty}
              </span>
              <button onClick={() => removeItem(item.id)}>‚ùå</button>
            </div>
          ))}
        </div>

        <div id="total">Total: {total.toFixed(2)}</div>
        <button id="checkout-btn">Checkout</button>
      </div>
    </div>
  );
}
